public class Hero {

    String image;
    String name, team;

    public Hero(String image, String name, String team) {
        this.image = image;
        this.name = name;
        this.team = team;
    }

    public String getImage() {
        return image;
    }

    public String getName() {
        return name;
    }

    public String getTeam() {
        return team;
    }
}
